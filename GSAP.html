GSAP 動畫

GSAP 官網
中文文本

在<head>加入:

<!-- GSAP -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.11.4/gsap.min.js"></script>

效果展示

gsap.to()

<script>
  document.getElementById('animate-btn1').addEventListener('click', () => {
    gsap.to("#box1", {
      x: 200,
      rotation: 360,
      duration: 2,
      ease: "power2.inOut"
    });
  });
</script>

gsap.from()

<script>
  document.getElementById('animate-btn2').addEventListener('click', () => {
    gsap.from("#box2", {
      x: 200,
      rotation: 360,
      duration: 2,
      ease: "power2.inOut"
    });
  });
</script>

gsap.fromTo()

<script>
  document.getElementById('animate-btn3').addEventListener('click', () => {
    gsap.fromTo("#box3", {
      x: 200,
      rotation: -180,
      duration: 2,
      ease: "power2.inOut"
    },
    {
      x: 200,
      rotation: 180,
      duration: 2,
      ease: "power2.inOut"
    });
  });
</script>
