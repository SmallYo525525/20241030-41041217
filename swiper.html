Swiper 幻燈片

Swiper 官網

範例程式碼

在<head>引入與樣式設定

<!-- swiper -->
<link rel="stylesheet" href="https://unpkg.com/swiper/swiper-bundle.min.css">
<style>
    .swiper {
        width: 100%;
        height: 300px;
    }
</style>

Swiper 幻燈片範例

<div class="container mt-5 text-center">
    <div class="swiper">
        <div class="swiper-wrapper">
            <div class="swiper-slide"><img src="http://fakeimg.pl/300x250/?text=1" alt=""></div>
            <div class="swiper-slide"><img src="http://fakeimg.pl/300x250/?text=2" alt=""></div>
            <div class="swiper-slide"><img src="http://fakeimg.pl/300x250/?text=3" alt=""></div>
            <div class="swiper-slide"><img src="http://fakeimg.pl/300x250/?text=4" alt=""></div>
            <div class="swiper-slide"><img src="http://fakeimg.pl/300x250/?text=5" alt=""></div>
        </div>
        <!-- 添加導航按鈕 -->
        <div class="swiper-button-next"></div>
        <div class="swiper-button-prev"></div>
        <!-- 添加滾動條 -->
        <div class="swiper-scrollbar"></div>
        </div>
</div>

<script src="https://unpkg.com/swiper/swiper-bundle.min.js"></script>
<script>
    const swiper = new Swiper('.swiper', {
        slidesPerView: 1,
        spaceBetween: 10,
        navigation: {
            nextEl: '.swiper-button-next',
            prevEl: '.swiper-button-prev',
        },
        scrollbar: {
            el: '.swiper-scrollbar',
            hide: true,
        },
    });
</script>
